m = mecab.MeCab()

# very basic preprocessing
# remove josa, hada/toeda forms, verb-final da, punctuation and quotes
# remove some basic words and characters that remain
# also, standardize 3-digit or below numbers to "#"

exclude_pos = ["JKS", "JKC", "JKG", "JKO", "JKB", "JKV", "JKQ", "JC", "JX", "EC", "XSV+EC", "XSV+ETM", "XSV+EP", "EF", "SC", "SF", "SSC", "SSO"]
exclude_tok = ["", "", "", "", "", "", "", "", "", "", ".", "-", '"', "", ""]

def tfidf_preprocess(doc_list):
    doc_toks = []
    for sent in doc_list:
        pos = m.pos(sent)
        for tup in pos:
            if tup[0] not in exclude_tok and tup[1] not in exclude_pos:
                doc_toks.append(re.sub(r"[0-9]{1,3}", "#", tup[0]).upper())
    return " ".join(doc_toks)

def find_most_similar(embeddings, article_id):
    sim_scores = list(cosine_similarity(embeddings[article_id:article_id+1, :], embeddings)[0])
    sim_pairs = sorted([(scr, idx) for idx, scr in enumerate(sim_scores)], key=lambda x:x[0], reverse=True)
    most_sim_score, most_sim = sim_pairs[1]
    print("source article:")
    print(" ".join(articles[article_id]["article_original"]))
    print("\nmost similar: {:.3f}".format(most_sim_score))
    print(" ".join(articles[most_sim]["article_original"]))
    return most_sim_score, most_sim

further projection and plotting¶

we can use UMAP to project this data down even further into 2D space. we could use SVD/PCA for this as well, or something like t-SNE, but UMAP seems like a newer off-the-shelf go-to for text data.

from the UMAP-learn package:

UMAP is a fairly flexible non-linear dimension reduction algorithm. It seeks to learn the manifold structure of your data and find a low dimensional embedding that preserves the essential topological structure of that manifold.

we can then try to group the projected points using clustering. we just take bayesian GMM as first-pass try.

we can specify cluster number; in order to try to subdivide the articles roughly, we specify 20 groups.
